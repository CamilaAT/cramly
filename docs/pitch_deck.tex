% ---------- Founder ----------
\begin{frame}{Founder \& Founder-Market Fit}
  \begin{itemize}
    \item Soy estudiante de la Universidad del Pacífico — \textbf{vivo el problema cada ciclo}.
    \item Llevo 5–7 cursos por semestre, cada uno con su sílabo en PDF.
    \item Conozco el dolor de primera mano: parciales olvidados, semanas saturadas, no saber cuánto necesito para aprobar.

% ---------- Problema ----------
\begin{frame}{El problema}
  Los universitarios reciben \textbf{5–7 sílabos en PDF} al inicio del ciclo. Las fechas, pesos y cronogramas están dispersos en tablas largas y documentos distintos.
  \vspace{0.3cm}
  \begin{itemize}
    \item Se olvidan entregas y parciales (aunque estén en el sílabo).
    \item No anticipan las semanas donde se juntan varias evaluaciones.
    \item No recuerdan cuánto pesa cada cosa ni cuánto necesitan para aprobar.
    \item Los cambios de fecha que el profe dice en clase se pierden.
    \item El sílabo queda como un \textbf{PDF muerto} que nadie reabre.
  \end{itemize}
  \vspace{0.2cm}
  \footnotesize\textit{Validado con 5 estudiantes de la UP.}
\end{frame}

% ---------- Solución & Insight ----------
\begin{frame}{Solución \& Insight}
  \begin{block}{Insight}
    El sílabo no debería leerse una vez. Debería convertirse \textbf{automáticamente} en un sistema vivo de planificación.
  \end{block}
  \vspace{0.2cm}
  El estudiante sube sus sílabos y obtiene en minutos:
  \begin{itemize}
    \item Tabla consolidada de evaluaciones (fechas + pesos).
    \item Gráfico de carga por semana y \textbf{semanas críticas}.
    \item Calendario de mes con clases y evaluaciones.
    \item Calculadora de nota final (escala 0–20).
    \item Chat para complementar lo que el profe dice en clase.
    \item Export a Google Calendar (.ics).
  \end{itemize}
\end{frame}

% ---------- Arquitectura ----------
\begin{frame}{Arquitectura}
  \begin{center}
    PDF(s) $\rightarrow$ \textbf{Claude (Sonnet 4.6)} lectura nativa + structured outputs\\[0.2cm]
    $\rightarrow$ JSON (cursos, pesos, fechas, horario)\\[0.2cm]
    $\rightarrow$ Normalización de fechas $\rightarrow$ Score de carga\\[0.2cm]
    $\rightarrow$ Dashboard · Calendario · Calculadora · Export .ics
  \end{center}
  \vspace{0.3cm}
  \footnotesize
  \textbf{Herramientas del curso:} Document AI con Claude · structured outputs · Streamlit.\\
  \textbf{Stack:} Python · Streamlit · Claude API · Plotly · icalendar.
\end{frame}

% ---------- Why now ----------
\begin{frame}{Why now?}
  \begin{itemize}
    \item Los LLMs (Claude) ahora leen PDFs complejos con tablas \textbf{de forma nativa y confiable} — antes no era posible extraer cronogramas con esta precisión.
    \item Los \textbf{structured outputs} hacen la extracción robusta y barata.
    \item Estudiantes 100\% digitales, acostumbrados a apps de productividad.
  \end{itemize}
\end{frame}

% ---------- Competencia & Moat ----------
\begin{frame}{Competencia \& Moat}
  \begin{itemize}
    \item \textbf{Hoy:} agenda a mano, Notion/Excel manual, recordatorios del REA, Google Calendar a mano.
    \item Ninguno \textbf{parte del PDF del sílabo} ni entiende formatos variados.
    \item \textbf{Moat:} extracción afinada para sílabos LatAm (formatos, fórmulas de notas, tablas), + datos de uso que mejoran el producto, + integración con flujo del estudiante.
